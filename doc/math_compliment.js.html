/*--------------------------- Calculate Functions ----------------------------*/

/**
 * Compliments the values in the array.
 * @param {number[]} array - the input array
 * @returns {number[]} the result of the operation
 */
Compliment.prototype.calculate = function(array) {

   var result = array.slice();

   for(var i = 0; i < array.length; i++) {
      result[i] = ~result[i];
   }

   return result;

}

/**
 * Compliments the values between the indices.
 * @param {number[]} array - the input array
 * @param {number} begin - the beginning index
 * @param {number} end - the ending index
 * @returns {number[]} the result of the operation
 */
Compliment.prototype.calculateRange = function(array, begin, end) {

   var result = array.slice();

   for(var i = begin; i < end && i < result.length; i++) {
      result[i] = ~result[i];
   }

   return result;
}
